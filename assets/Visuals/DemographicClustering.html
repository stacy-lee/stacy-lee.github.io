from sklearn.cluster import DBSCAN

# Apply DBSCAN
db = DBSCAN(eps=9, metric='euclidean', min_samples=10)
db.fit(x)

# Get cluster labels and assign plotting colors/labels.
dblbls = set(db.labels_)
dbclrs = sns.hls_palette(len(dblbls))
dbcls = ['Class {0}'.format(idx) if idx >= 0 else 'Noise' for idx in dblbls]
cols = ['PCA1','PCA2','political_affiliation']
dt=pd.DataFrame(data= {'PCA1': x[:,0],'PCA2': x[:,1],'political_affiliation':y})
pc = pd.DataFrame(np.concatenate((x, db.labels_.reshape((743, 1))), axis=1), columns = cols)

# Make plot
sns.set(font_scale=2.0)
fig, ax = plt.subplots(figsize=(12, 10))

# Plot known clusters
for idx in range(len(lbls)):
    tmp_df = dt[dt['political_affiliation'] == idx]
    ax.scatter(tmp_df['PCA1'], tmp_df['PCA2'], color=clr[idx], label=lbls[idx], alpha=0.2, s=180)

# Plot DBSCAN clusters (and noise)
for idx in list(dblbls):
    tmp_pdf = pc[pc['political_affiliation'] == idx]
    ax.scatter(tmp_pdf['PCA1'], tmp_pdf['PCA2'], color=dbclrs[idx], label=dbcls[idx], alpha=1, s=30)

ax.set_xlim(-50, 80)
ax.set_ylim(-50, 80)
ax.set_xlabel('PCA 1')
ax.set_ylabel('PCA 2')
ax.set_title('Year Built and Customer Age w.r.t. Political Affiliation')
ax.legend(bbox_to_anchor=(1, 1), loc=2)

sns.despine(offset=5, trim=True)
sns.set(font_scale=1.0)
